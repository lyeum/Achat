```python
@Slot(result=str)
def getDefaultWorld(self) -> str:
    """세계관 목록에서 첫 번째 world_id / scenario_id / act_id를 반환."""
    worlds = json.loads(self.getWorldList())
    if not worlds:
        return json.dumps({})
    w = worlds[0]
    sc = w["scenarios"][0] if w.get("scenarios") else {}
    act = sc["acts"][0] if sc.get("acts") else {}
    return json.dumps({
        "world_id": w["world_id"],
        "scenario_id": sc.get("scenario_id", ""),
        "act_id": act.get("act_id", ""),
    }, ensure_ascii=False)
```

#### 4-2-2. 세계관 RAG 탭

**필요 신규 슬롯:** `bridge.getWorldKnowledgeDB()`

```python
@Slot(result=str)
def getWorldKnowledgeDB(self) -> str:
    """world_knowledge 컬렉션 전체 청크를 반환."""
    import chromadb
    cfg = getattr(self._agent, "cfg", {})
    chroma_path = cfg.get("chroma_path", "./chroma_dev")
    client = chromadb.PersistentClient(path=chroma_path)
    try:
        col = client.get_collection("world_knowledge")
        result = col.get(include=["documents", "metadatas"])
        chunks = []
        for i, doc in enumerate(result["documents"]):
            meta = result["metadatas"][i] if result["metadatas"] else {}
            chunks.append({"id": result["ids"][i], "content": doc, "source": meta.get("source", "")})
        return json.dumps({"total": len(chunks), "chunks": chunks}, ensure_ascii=False)
    except Exception as e:
        return json.dumps({"total": 0, "chunks": [], "error": str(e)}, ensure_ascii=False)
```

QML: source 파일명 기준 그룹화, 청크 content 읽기 전용 표시.  
"재인덱싱" 버튼 → `bridge.reindexWorldKnowledge()` 슬롯 호출.

```python
@Slot(result=bool)
def reindexWorldKnowledge(self) -> bool:
    """rag/sources/world/ 파일을 강제 재인덱싱한다."""
    from rag.index import index_world
    cfg = getattr(self._agent, "cfg", {})
    rag_dir = Path(cfg.get("rag_world_dir", "./rag/sources/world"))
    chroma_path = cfg.get("chroma_path", "./chroma_dev")
    embedding_model = cfg.get("embedding_model", "BAAI/bge-m3")
    try:
        index_world(rag_dir, chroma_path, embedding_model, force=True)
        return True
    except Exception:
        return False
```

## 항목 6. 세션 구분 — SettingsPanel 목록 UI + switchSession 슬롯

### 6-1. 현재 구조

- `SessionManager.list_sessions(char_id)` → `bridge.listSessions(char_id)` 이미 존재
- `SessionState`에 `world_id` 필드 이미 있음 (`Optional[str] = None`)
- `switchSession` 슬롯 없음 — 새로 추가 필요

세션 인덱스: `data/sessions/{char_id}/sessions.json`  
세션 상태: `data/sessions/{char_id}/{session_id}/state.json`

### 6-2. switchSession Slot 추가

**파일:** `ui_ux/bridge.py`

```python
@Slot(str, result=bool)
def switchSession(self, session_id: str) -> bool:
    """해당 session_id로 세션을 전환한다.

    - 현재 세션을 저장
    - SessionManager.activate(char_id, session_id) 호출
    - 에이전트 세션 상태 갱신
    - 최근 10턴 대화 로그 복원 (messageAdded emit)
    """
    char_id = (self._agent.character or {}).get("id", "")
    if not char_id or not session_id:
        return False

    # 현재 세션 저장
    if self._agent.session is not None:
        self._session_manager.save_state(self._agent.session)

    # 세션 전환
    new_state = self._session_manager.activate(char_id, session_id)
    if new_state is None:
        return False

    self._agent.session = new_state

    # 최근 10턴 로그 복원은 dialogue_log가 디스크에 없으므로
    # "세션 전환됨" 시스템 메시지만 emit
    self.messageAdded.emit("system", f"세션 '{session_id[:14]}' 로 전환했습니다.")
    self.affectionChanged.emit(new_state.affection)
    self.moodChanged.emit(new_state.mood)
    return True
```

> **참고:** 현재 `dialogue_log`는 런타임 인메모리만 유지 → 세션 전환 시 대화 복원은 불가.  
> 중기 과제로 `dialogue_log`의 최근 10턴을 `state.json`에 함께 저장하는 구조 추가 검토.

```python
MAX_SESSIONS = 3

def _create_session(self, char_id: str) -> SessionState:
    # 생성 전 초과 세션 정리
    index = self._load_index(char_id)
    if len(index) >= self.MAX_SESSIONS:
        oldest = min(index, key=lambda x: x.get("last_active", ""))
        self._evict_session(char_id, oldest["session_id"])
    ...

def _evict_session(self, char_id: str, session_id: str) -> None:
    """세션 디렉토리와 인덱스 항목을 제거한다."""
    import shutil
    session_dir = self._char_dir(char_id) / session_id
    if session_dir.exists():
        shutil.rmtree(session_dir)
    index = self._load_index(char_id)
    index = [i for i in index if i["session_id"] != session_id]
    self._save_index(char_id, index)
```
